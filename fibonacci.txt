FIBONACCI:
  MOV n 3
  
  CMP n 0
  JE RETURN_0
  CMP n 1
  JE RETURN_0
  CMP n 2
  JE RETURN_1

  MOV prev 0
  MOV prox 1
  
  MOV i 0
  LOOP:
    CMP i n
    JE END_LOOP
    
    MOV aux prox
    ADD prox prev
    MOV prev aux
    
    ADD i 1
    JMP LOOP

  END_LOOP:
    MOV return prox
    JMP END

  RETURN_0:
    MOV return 0
    JMP END

  RETURN_1:
    MOV return 1
    JMP END

END:
